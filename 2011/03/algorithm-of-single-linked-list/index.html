单链表的一些算法实现

Thu 10 March 2011 by Zhaoyu Li

记得数据结构课上老师讲过单链表的一些算法实现，后来渐渐意识到这一部分非常重要，今天在这里搜集整理一下。

首先定义节点：

struct Node{
    int data; //数据
    Node* next;//下一个结点
};

创建带n个数据的单链表：

void LinkList::Init(){
    head = new Node();
    head-&gt;next = NULL;
}

void LinkList::Create(int n){
    Node* p = head;
    for (int i = 1 ; i <= n; ++i)  {
        Node* temp = new Node();
        temp->data = i ;
        temp->next =NULL;
        p->next = temp;
        p = temp;
    }
}

在索引为i的位置插入数据为data的结点：

bool LinkList::Insert(int i, int data) {
    if (i &lt;0 || i &gt; GetLen()){
        return false;
    }
    Node* p = head;
    Node* q = head;
    for (int j = 0; j&lt;i; j++){
        q = p-&gt;next;
        p = p-&gt;next;
    }
    Node* InsertNode = new Node;
    InsertNode-&gt;data = data;
    InsertNode-&gt;next = q-&gt;next;
    q-&gt;next = InsertNode;
    return true;
}

删除索引为i的结点：

bool LinkList::Delete(int i){
    if (i &lt; 0 || i &gt;= GetLen()) {
        return false;
    }
    Node* p = head;
    Node* q  = head;
    for (int j = 0; j&lt;i; j++){
        q = p-&gt;next;
        p = p-&gt;next;
    }
    Node* DelNode = q-&gt;next;
    q-&gt;next = DelNode-&gt;next;
    delete DelNode;
    return true;
}

归并单链表：

void LinkList::Merge(LinkList&amp; list){
    Node* p = head-&gt;next;
    while (p-&gt;next){
        p = p-&gt;next;
    }
    Node* listNode = list.GetNode();
    p-&gt;next = listNode-&gt;next;
}

就地反转：

void LinkList::Back(){
    /*先把原来的链表头结点断开
    *建立一个新的链表，然后把原来的链表插入到新链表的头部
    *最后把原来链表头部指向新链表，最后注意要释放新链表的头部
    *也就是最后一个元素**************************************/
    LinkList tempList;
    tempList.Init();
    Node* TempNode = tempList.GetNode();
    Node* n = TempNode;
    Node* p = head-&gt;next;
    while(p){
        Node* q = p-&gt;next;
        Node* temp = p;
        temp-&gt;next = TempNode;
        temp-&gt;data = p-&gt;data;
        TempNode = temp;
        p = q; 
    }
    head-&gt;next = TempNode;
    Delete(GetLen() - 1);
}

是否有环：

bool LinkList::HaveCircle(){
    Node* p = head;
    Node* q = head;
    while ((p-&gt;next != NULL) &amp;&amp; (q-&gt;next != NULL) &amp;&amp; (q-&gt;next-&gt;next != NULL)){
        p = p-&gt;next;
        q = q-&gt;next-&gt;next;
        if (p == q){
            return true;
        }
    }
    return false;
}

选择排序：

void LinkList::Sort() {
    Node* p = head-&gt;next;
    while(p){
        Node* small = p;
        Node* q = p-&gt;next;
        while(q){
            if (q-&gt;data &lt; small-&gt;data) {
                small = q;
            }
        q = q-&gt;next;
        }
        if (p != small){
            int temp;
            temp = p-&gt;data;
            p-&gt;data = small-&gt;data;
            small-&gt;data = temp;
        }
    p = p-&gt;next;
    }
}
